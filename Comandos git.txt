git checkout master

# Me uno al repo remoto
git remote add upstream https://github.com/HotCode2025/Debuggers-UTN-Segundo-Semestre.git

# listo todas las ramas del repo remoto
git fetch 

# Descarga los datos de la rama Proyecto_Java desde el remoto 'equipo'

git clone --branch Proyecto_Java --single-branch https://github.com/HotCode2025/Debuggers-UTN-Segundo-Semestre.git repo
